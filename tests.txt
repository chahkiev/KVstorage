POST(add key-value):
url = 'http://localhost:8000/kv/'
requestData = {"key" : "zero", "value" : {"coca-cola": "bar", "aaa" : "bbb"}}
jsn = json.dumps(requestData)
r = requests.post(url, data  = jsn )


PUT(change value):
url = 'http://localhost:8000/kv/key2/'
requestData = {"value" : {"PUUUUUT": "bar", "aaa" : "bbb"}}
jsn = json.dumps(requestData)
r = requests.put(url, data  = jsn )


GET(get key-value):
url = 'http://localhost:8000/kv/key2/'
r = requests.get(url)


DELETE(delete key-value):
url = 'http://localhost:8000/kv/key777/'
r = requests.delete(url)
